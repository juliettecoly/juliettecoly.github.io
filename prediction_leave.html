In [12]:
# Scale, Create a Pipeline, Choose hyperparameter
pipelines = {"l1": make_pipeline(StandardScaler(), 
                                 LogisticRegression(penalty = "l1",solver = "liblinear")),
            "l2" : make_pipeline(StandardScaler(), LogisticRegression()),
            "rf": make_pipeline(StandardScaler(), RandomForestClassifier()),
            "gb": make_pipeline(StandardScaler(), GradientBoostingClassifier())}
ll = [1,2,5,10]
hyperparameters = {"l1" : {'logisticregression__C':ll}, 
                   "l2" : {'logisticregression__C':ll}, 
                    "rf" : {'randomforestclassifier__min_samples_leaf' : ll},
                    "gb" : {'gradientboostingclassifier__min_samples_leaf' : ll}}
In [13]:
# Fit the models
fitted_models = {}
for m in pipelines.keys():
    model = GridSearchCV(pipelines[m], hyperparameters[m])
    fitted_models[m] = model.fit(X_train, y_train)
